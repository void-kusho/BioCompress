```c
#ifndef BIOCOMPRESS_PARSER_H
#define BIOCOMPRESS_PARSER_H

#include <stdint.h>
#include <stddef.h>
#include <stdbool.h>
#include "biocompress.h"

// Quality encoding schemes
typedef enum {
    QUAL_ENCODING_SANGER,      // Phred+33 (0-93)
    QUAL_ENCODING_SOLEXA,      // Phred+64 (5-62) 
    QUAL_ENCODING_ILLUMINA_13,  // Phred+64 (0-62) Illumina 1.3-1.8
    QUAL_ENCODING_ILLUMINA_18,  // Phred+64 (0-62) Illumina 1.8+
    QUAL_ENCODING_AUTO         // Auto-detect
} quality_encoding_t;

// Record structures
typedef struct {
    char* header;       // NULL-terminated header line (without >)
    char* sequence;    // NULL-terminated sequence (ACGTUN only)
    size_t seq_len;
} fasta_record_t;

typedef struct {
    char* header;
    char* sequence;
    char* quality;      // Quality scores as string
    size_t seq_len;
    quality_encoding_t encoding;
} fastq_record_t;

// Parser state for streaming
typedef struct parser_state parser_state_t;

// FASTA functions
parser_state_t* fasta_open(const char* filepath);
void fasta_close(parser_state_t* ctx);
int fasta_next_record(parser_state_t* ctx, fasta_record_t* record);
void fasta_record_free(fasta_record_t* record);

// FASTQ functions
parser_state_t* fastq_open(const char* filepath);
void fastq_close(parser_state_t* ctx);
int fastq_next_record(parser_state_t* ctx, fastq_record_t* record);
void fastq_record_free(fastq_record_t* record);

// Validation
bool biocompress_validate_sequence(const char* seq, size_t len);
int biocompress_detect_quality_encoding(const char* quality, size_t len);
int biocompress_decode_quality(const char* quality, size_t len, 
                               int8_t* output, quality_encoding_t encoding);

#endif
```

Create `src-c/src/fasta.c`:

1. `fasta_open()`: Opens file, validates magic (first char '>'), returns parser_state
2. `fasta_next_record()`: 
   - Skips to next '>' line
   - Extracts header (content after '>')
   - Accumulates sequence lines until next '>' or EOF
   - Validates sequence with `biocompress_validate_sequence()` per D-07
   - Returns record with streaming (one-at-a-time) per D-05
3. `fasta_record_free()`: Frees header and sequence buffers
4. `biocompress_validate_sequence()`: Rejects any char not in ACGTUN per D-07

Use memory-efficient streaming - allocate buffers as needed, never load entire file.

Create `src-c/src/fastq.c`:

1. `fastq_open()`: Opens file, validates first line starts with '@'
2. `fastq_next_record()`:
   - Reads @ header line
   - Reads sequence line
   - Reads + separator line  
   - Reads quality line
   - Validates sequence with biocompress_validate_sequence() per D-07
   - Matches sequence length equals quality length
3. `biocompress_detect_quality_encoding()`:
   - Analyze quality string character range:
     - If all chars in range '!' to 'M' (0-50): SANGER (Phred+33)
     - If chars in range ';' to 'o' (59-111): SOLEXA (Phred+64)
     - If all chars in range '@' to 'o' (64-111): Illumina 1.3-1.8 (Phred+64)
     - Otherwise: Illumina 1.8+ (Phred+64)
   - Return detected encoding per D-06
4. `biocompress_decode_quality()`: Convert ASCII quality to integer Phred scores

Handle all quality encodings per D-06:
- SANGER: Phred+33 (ASCII 33-126, values 0-93)
- SOLEXA: Phred+64 (ASCII 59-111, values 5-62) — deprecated but supported
- Illumina 1.3-1.8: Phred+64 (ASCII 64-111, values 0-62)
- Illumina 1.8+: Phred+64 (ASCII 66-126, values 0-93)
