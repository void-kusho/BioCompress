#ifndef BIOCOMPRESS_HUFFMAN_H
#define BIOCOMPRESS_HUFFMAN_H

#include <stdint.h>
#include <stddef.h>
#include "biocompress.h"

#define HUFFMAN_SYMBOLS 4  // A, C, G, T (2-bit)

// Canonical Huffman tree for DNA (fixed alphabet per D-08)
typedef struct {
    uint8_t code[HUFFMAN_SYMBOLS];      // Canonical codes
    uint8_t code_len[HUFFMAN_SYMBOLS];   // Code lengths (1-8 bits)
    uint16_t num_codes;                   // Number of codes
} huffman_code_t;

typedef struct huffman_ctx huffman_ctx_t;

// Build canonical Huffman code from frequencies
huffman_ctx_t* huffman_build(const uint32_t* frequencies, int num_symbols);

// Encode symbol sequence to bitstream
int huffman_encode(huffman_ctx_t* ctx, 
                  const uint8_t* symbols, size_t num_symbols,
                  uint8_t** output, size_t* output_size);

// Decode bitstream to symbol sequence  
int huffman_decode(huffman_ctx_t* ctx,
                  const uint8_t* input, size_t input_size,
                  uint8_t** output, size_t* output_size);

void huffman_free(huffman_ctx_t* ctx);
```

Create `src-c/src/huffman.c`:

1. **Symbol mapping per D-08:**
   - A = 0 (binary 00)
   - C = 1 (binary 01)
   - G = 2 (binary 10)
   - T = 3 (binary 11)
   - Map ACGTUN to these 4 symbols (N -> A, U -> T for simplicity)

2. **Canonical construction per D-09:**
   - Build frequency table from input
   - Create canonical Huffman codes (sorted by frequency, then lexicographically)
   - Store as code + code_len pairs (more compact than full tree)

3. **Encode:** Replace each symbol with its code/bits
4. **Decode:** Use code_len to determine symbol boundaries, look up in table

#ifndef BIOCOMPRESS_ANS_H
#define BIOCOMPRESS_ANS_H

#include <stdint.h>
#include <stddef.h>
#include "biocompress.h"

#define ANS_STATE_BITS 32    // 32-bit state per D-11
#define ANS_TABLE_BITS 12    // Table size = 2^12 = 4096

typedef enum {
    ANS_VARIANT_TANS,   // Table-based ANS (faster) per D-10
    ANS_VARIANT_RANS    // Range ANS (better ratio) per D-10
} ans_variant_t;

typedef struct ans_ctx ans_ctx_t;

// Initialize ANS encoder/decoder
ans_ctx_t* ans_init(ans_variant_t variant, int compression_level);

// Encode symbol sequence
int ans_encode(ans_ctx_t* ctx,
              const uint8_t* input, size_t input_size,
              uint8_t** output, size_t* output_size);

// Decode back to original
int ans_decode(ans_ctx_t* ctx,
               const uint8_t* input, size_t input_size,
               uint8_t** output, size_t* output_size);

void ans_free(ans_ctx_t* ctx);
```

Create `src-c/src/ans.c`:

1. **tANS implementation:**
   - Use canonical ANS table (pre-computed)
   - State = 32-bit per D-11
   - Encode: update state with symbol frequency normalization
   - Decode: inverse update with table lookup

2. **rANS implementation:**
   - Range-based ANS (more efficient for larger alphabets)
   - Divide state range by symbol probability
   - More complex but better compression ratio

3. **Variant selection per D-10:**
   - Level 1-2: Huffman (handled by huffman module)
   - Level 3-4: tANS (faster, good ratio)
   - Level 5: rANS (best ratio, slower)

4. Use 32-bit state word size per D-11
